## Usage

```jsx
import { Label, Input } from '@sozialhelden/ui';
import { useId } from 'react';

export function App() {
  const id = useId();
  return (
    <div>
      <Label htmlFor={id}>Your email address</Label>
      <Input id={id} type="email" />
    </div>
  );
}
```
