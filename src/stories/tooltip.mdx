## Usage

```jsx
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from '@sozialhelden/ui';

export function App() {
  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger>Hover</TooltipTrigger>
        <TooltipContent>
          Content of the tooltip
        </TooltipContent>
      </Tooltip>
    </TooltipProvider>
  );
}
```
