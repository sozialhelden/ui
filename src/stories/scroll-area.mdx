## Usage

```jsx
import { ScrollArea } from '@sozialhelden/ui';

export function App() {
  return (
    <ScrollArea>
      Scrollable content goes here.
    </ScrollArea>
  );
}
```

## Scroll to specific position

```jsx
import { useRef } from "react";
import { ScrollArea, ScrollAreaRef } from '@sozialhelden/ui';

export function App() {
  const scrollArea = useRef<ScrollAreaRef>(null);

  return (
    <>
        <button onClick={() => {
            scrollArea.current?.scrollTo({ top: 100, left: 0, behavior: 'smooth' });
        }}>
          Scroll me baby
        </button>
        <ScrollArea ref={scrollArea}>
          Scrollable content goes here.
        </ScrollArea>
    </>
  );
}
```
