## Usage

```jsx
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from '@sozialhelden/ui';

export function App() {
  return (
    <Popover>
      <PopoverTrigger>Open popover</PopoverTrigger>
      <PopoverContent>
          Hello world.
      </PopoverContent>
    </Popover>
  );
}
```
