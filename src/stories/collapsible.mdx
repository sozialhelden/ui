## Usage

```jsx
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from '@sozialhelden/ui';

export function App() {
  return (
    <Collapsible>
      <CollapsibleTrigger>Can I use this in my project?</CollapsibleTrigger>
      <CollapsibleContent>
        Yes. Free to use for personal and commercial projects. No attribution required.
      </CollapsibleContent>
    </Collapsible>
  );
}
```
