## Usage

```jsx
import { Checkbox, Label } from '@sozialhelden/ui';
import { useId } from "react";

export function App() {
  const id = useId();
  return (
    <div className="flex gap-2">
        <Checkbox id={id} />
        <Label htmlFor={id}>Checkbox</Label>
    </div>
  );
}
```
