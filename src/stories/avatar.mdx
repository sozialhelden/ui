## Usage

```jsx

export function App() {
  return (
    <Avatar>
      <AvatarImage
        src="https://github.com/id-Software.png"
        alt="@id-Software"
      />
      <AvatarFallback>ID</AvatarFallback>
    </Avatar>
  );
}
```
